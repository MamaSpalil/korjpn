int GetPercent(int MaxState,int State)
{

	int PercentSet;
	int PercentState;
	if(State || MaxState != 0)
	{
	PercentSet = MaxState/100;
	PercentState = State/PercentSet;
	if(PercentState > 100)
    PercentState = 100;
	else if (PercentState <= 0)
	PercentState = 1;
	if((PercentState == 100)&&(MaxState != State))
    PercentState = 99;

	return PercentState;
	}
	else
	{
    PercentState = 0;
	return PercentState;
	}
}

void __declspec(naked) LoadLifeShow()
{
    _asm
    {
	   push HPCurrentShow
       mov MovHP_OffSet, 0x000000000
       jmp MovHP_OffSet //

    }
}
