void cHookHP_Show()
{
	//SET PUSH TO LIFE — expand mov ax,[ecx+0x22] to mov eax,[ecx+0x22];nop at 0x59F80E
	SetByte( (PVOID) 0x0059F80E, 0x8B);
	SetByte( (PVOID) 0x0059F80F, 0x41);
	SetByte( (PVOID) 0x0059F810, 0x22);
	SetByte( (PVOID) 0x0059F811, 0x90);

	//SET PUSH TO LIFE 2 — expand mov dx,[ecx+0x26] to mov edx,[ecx+0x26];nop at 0x59F80A
	SetByte( (PVOID) 0x0059F80A, 0x8B);
	SetByte( (PVOID) 0x0059F80B, 0x51);
	SetByte( (PVOID) 0x0059F80C, 0x26);
	SetByte( (PVOID) 0x0059F80D, 0x90);
	//SET PUSH TO Mana — expand mov ax,[ecx+0x24] to mov eax,[ecx+0x24];nop at 0x59F82E
	SetByte( (PVOID) 0x0059F82E, 0x8B);
	SetByte( (PVOID) 0x0059F82F, 0x41);
	SetByte( (PVOID) 0x0059F830, 0x24);
	SetByte( (PVOID) 0x0059F831, 0x90);

	//SET PUSH TO Mana 2 — expand mov dx,[ecx+0x28] to mov edx,[ecx+0x28];nop at 0x59F828
	SetByte( (PVOID) 0x0059F828, 0x8B);
	SetByte( (PVOID) 0x0059F829, 0x51);
	SetByte( (PVOID) 0x0059F82A, 0x28);
	SetByte( (PVOID) 0x0059F82B, 0x90);
	////Stamina Coordinats — adjust Stamina bar X position bytes at push 0x43BD8000 (0x59FDC9)
	SetByte( (PVOID) 0x0059FDCB, 0x80);
	SetByte( (PVOID) 0x0059FDCC, 0xBD);
	////SD  Coordinats — adjust SD bar X position bytes at push 0x43AE0000 (0x59FCEB)
	SetByte( (PVOID) 0x0059FCED, 0x00);//
	SetByte( (PVOID) 0x0059FCEE, 0xAE);//
	//XZ
	//SET PUSH TO Stamina — expand mov ax,[ecx+0x42] to mov eax,[ecx+0x42];nop at 0x59EB2C
	SetByte( (PVOID) 0x0059EB2C, 0x8B);
	SetByte( (PVOID) 0x0059EB2D, 0x41);
	SetByte( (PVOID) 0x0059EB2E, 0x42);
	SetByte( (PVOID) 0x0059EB2F, 0x90);

	//SET PUSH TO Shield — expand mov ax,[ecx+0x2C] to mov eax,[ecx+0x2C];nop at 0x59EE9C
	SetByte( (PVOID) 0x0059EE9C, 0x8B);
	SetByte( (PVOID) 0x0059EE9D, 0x41);
	SetByte( (PVOID) 0x0059EE9E, 0x2C);
	SetByte( (PVOID) 0x0059EE9F, 0x90);
	
	
	SetRange((LPVOID)0x0059ECFF, 6, ASM::NOP);		
    SetJmp((LPVOID)0x0059ECFF, LoadShieldShow);		
	SetRange((LPVOID)0x0059F06F, 6, ASM::NOP);		
    SetJmp((LPVOID)0x0059F06F, LoadStaminaShow);	
	SetRange((LPVOID)0x005A095D, 4, ASM::NOP);		
    SetJmp((LPVOID)0x005A095D, LoadLifeShow);		
	SetRange((LPVOID)0x005A0982, 4, ASM::NOP);		
    SetJmp((LPVOID)0x005A0982, LoadManaShow);		
}

int GetPercent(int MaxState,int State)
{

	int PercentSet;
	int PercentState;
	if(State || MaxState != 0)
	{
	PercentSet = MaxState/100;
	PercentState = State/PercentSet;
	if(PercentState > 100)
    PercentState = 100;
	else if (PercentState <= 0)
	PercentState = 1;
	if((PercentState == 100)&&(MaxState != State))
    PercentState = 99;

	return PercentState;
	}
	else
	{
    PercentState = 0;
	return PercentState;
	}
}

void __declspec(naked) LoadLifeShow()
{
    _asm
    {
	   push HPCurrentShow
       mov MovHP_OffSet, 0x005A0962
       jmp MovHP_OffSet // return after hooked mov ax,[ecx+0x22]+push eax

    }
}
void __declspec(naked) LoadManaShow()
{
    _asm
    {
	   push ManaCurrentShow
       mov MovMana_OffSet, 0x005A0987
       jmp MovMana_OffSet // return after hooked mov dx,[eax+0x24]+push edx

    }
}

void __declspec(naked) LoadStaminaShow()
{
    _asm
    {
	   push StaminaCurrentShow
	   push 0x43E98000
       mov MovStamina_OffSet, 0x0059F075
       jmp MovStamina_OffSet // return after hooked push esi+push 0x43E98000
    }
}
void __declspec(naked) LoadShieldShow()
{
    _asm
    {
		
	   push ShieldCurrentShow
	   push 0x43E98000
       mov MovShield_OffSet, 0x0059ED05
       jmp MovShield_OffSet // return after hooked push esi+push 0x43E98000
    }
}
